\textbf{Requirements:}
\begin{itemize}
	\item \textbf{Basic:} The character must already possess Magica Profana, Magica Divinitatis or Magica Naturae
	\item \textbf{Advanced:} None.
	\item \textbf{Expert:} None.
\end{itemize}
\textbf{Effects:}
\begin{itemize}
	\item \textbf{Untrained:} The character cannot enchant items.
	\item \textbf{Basic:} The character can put \textbf{Class C} enchantment on items.
	\item \textbf{Advanced:} The character can put \textbf{Class B} enchantment on items. The character can also put up to two enchantments on an item.
	\item \textbf{Expert:} The character can put \textbf{Class A} enchantment on items. The character can also put up to four enchantments on an item.
\end{itemize}\newpage
